# 자율개발 결과 조회

자율개발(`/자율개발`) 실행 후 결과를 빠르게 확인합니다.
빌드/테스트는 실행하지 않습니다 (read-only).

## 실행 절차

**단일 Bash 스크립트**로 모든 정보를 한 번에 출력합니다. (사용자 클릭 최소화)

### 통합 조회 스크립트

```bash
#!/bin/bash
set -e

# Arguments에서 세션 ID 추출 (없으면 최신)
SESSION_ID="${1:-}"
if [ -z "$SESSION_ID" ]; then
  RELAY_FILE=$(ls -t tmp/night-*-relay.md 2>/dev/null | head -1)
  if [ -z "$RELAY_FILE" ]; then
    echo "❌ 자율개발 결과가 없습니다."
    exit 0
  fi
  SESSION_ID=$(basename "$RELAY_FILE" | sed 's/^night-\(.*\)-relay\.md$/\1/')
fi

# 파일 경로
RELAY="tmp/night-${SESSION_ID}-relay.md"
BLOCKED="tmp/night-${SESSION_ID}-blocked.md"
RESULTS_DIR="tmp/night-${SESSION_ID}-results"
LOG="tmp/night-${SESSION_ID}-log.txt"

# 브랜치명 추출
BRANCH=$(grep -oP 'night/\d{8}-[a-z0-9-]+' "$RELAY" 2>/dev/null | head -1 || echo "unknown")

# ━━━ 헤더 ━━━
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo " 자율개발 결과 - ${SESSION_ID}"
echo " 브랜치: ${BRANCH}"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo ""

# ━━━ 릴레이 노트 ━━━
echo "## 📋 릴레이 노트"
echo ""
cat "$RELAY" 2>/dev/null || echo "릴레이 노트 없음"
echo ""

# ━━━ 블로커 ━━━
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "## 🚫 블로커"
echo ""
BLOCKED_LINES=$(wc -l < "$BLOCKED" 2>/dev/null || echo "0")
if [ "$BLOCKED_LINES" -le 2 ]; then
  echo "블로커 없음"
else
  cat "$BLOCKED"
fi
echo ""

# ━━━ 비용/턴 요약 ━━━
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "## 💰 태스크별 비용/턴 요약"
echo ""
python3 << PYEOF
import json, glob, os
total_cost = 0
results = []
for f in sorted(glob.glob('${RESULTS_DIR}/*.json')):
    try:
        d = json.load(open(f))
        name = os.path.basename(f).replace('.json','')
        status = d.get('subtype','unknown')
        turns = d.get('num_turns', 0)
        cost = d.get('total_cost_usd', 0)
        total_cost += cost
        results.append(f'  {name}: {status} ({turns}턴, \${cost:.2f})')
    except:
        pass
for r in results:
    print(r)
print(f'\\n  총 비용: \${total_cost:.2f}')
PYEOF
echo ""

# ━━━ 커밋 로그 ━━━
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "## 📝 커밋 로그"
echo ""
COMMITS=$(git log --oneline master.."$BRANCH" 2>/dev/null | head -10)
if [ -z "$COMMITS" ]; then
  echo "  커밋 없음"
else
  echo "$COMMITS"
fi
echo ""

# ━━━ 진행 상황 ━━━
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "## ⏳ 진행 상황"
echo ""
TMUX_SESSION=$(tmux list-sessions 2>/dev/null | grep -E "v103|retry|night" | head -1 || echo "")
if [ -n "$TMUX_SESSION" ]; then
  echo "✅ tmux 세션 실행 중"
  echo "$TMUX_SESSION"
  echo ""
  echo "최근 로그 (마지막 10줄):"
  tail -10 "$LOG" 2>/dev/null || echo "로그 없음"
else
  echo "✓ 자율개발 완료 (tmux 세션 종료됨)"
fi
echo ""

# ━━━ 다음 단계 ━━━
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "## 🎯 다음 단계"
echo ""
echo "  결과 만족 → /최종검수 로 빌드/테스트 재검증"
echo "  결과 불만족 → 브랜치 삭제 후 재작업"
echo "  모니터링 → tail -f $LOG"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
```

### 실행 방법

1. **Bash 도구로 위 스크립트를 단일 실행** (Arguments 전달)
2. 결과를 그대로 사용자에게 출력
3. 추가 질문이나 개입 없이 완료

### 여러 세션 처리

세션이 여러 개일 경우, 먼저 목록을 보여주고 사용자가 선택하도록 합니다:

```bash
ls -t tmp/night-*-relay.md | head -5
# 목록 출력 후 AskUserQuestion으로 선택
```

## 함정 (Pitfalls)

- **tmux 세션이 아직 실행 중일 수 있음** — "진행 상황" 섹션으로 확인, 실행 중이면 /최종검수 금지
- **result JSON 파싱 실패** — claude 비정상 종료 시 JSON이 깨짐, python3 파서가 except로 건너뜀
- **릴레이 노트에 태스크 결과가 누락될 수 있음** — 에이전트가 릴레이 작성을 빠뜨리면 자동 기록(git diff)만 남음
- **비용이 $0.00으로 나올 수 있음** — total_cost_usd 필드 없는 구버전 result JSON
